void solve() {
  int n;
  cin >> n;

  VI a(n + 10);
  For1(i, 1, n) { cin >> a[i]; }

  ll ans{};
  for (int i = 2; i <= n; i += 2) {
    int sum = a[i - 1] + a[i + 1];
    if (a[i] >= sum)
      continue;
    int g = sum - a[i];
    ans += g;
    a[i + 1] -= min(g, a[i + 1]);
  }
  cout << ans << '\n';
}
